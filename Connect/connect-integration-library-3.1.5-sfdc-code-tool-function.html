// Morgan & Morgan legal intake — replaces Lambda + chained SFDC Flows.
// Fires on call_analyzed. Branches on intake outcome extracted by post-call analysis.
const { call_id, analysis, contact } = event;
const { intake_outcome, case_type, jurisdiction, conflict_party, statute_status } = analysis;
// 1. Always log the AI call record
const aiCall = await sf.upsert('AI_Voice_Call__c', 'Voice_Provider_Record_ID__c', {
Voice_Provider_Record_ID__c: call_id,
AI_Call_Summary__c: analysis.summary,
Intake_Outcome__c: intake_outcome,
Contact__c: contact.sfdc_id,
});
// 2. Branch on intake outcome
if (intake_outcome === 'qualified') {
const matter = await sf.create('Matter__c', {
Contact__c: contact.sfdc_id, Case_Type__c: case_type, Jurisdiction__c: jurisdiction,
Status__c: 'New', AI_Voice_Call__c: aiCall.Id,
});
await sf.invokeFlow('Start_Retainer_Process', { matterId: matter.Id });
await slack.send('#intake-qualified', `New matter: ${case_type} (${jurisdiction}) — ${contact.name}`);
}
else if (intake_outcome === 'conflict') {
await sf.invokeFlow('Conflict_Clearance_Workflow', { contactId: contact.sfdc_id, conflictParty: conflict_party });
await slack.send('#conflicts-review', `⚠️ Conflict flagged: ${contact.name} vs. ${conflict_party}`);
}
else if (intake_outcome === 'out_of_jurisdiction') {
await sf.invokeFlow('Notify_Partner_Firm', { contactId: contact.sfdc_id, jurisdiction });
}
else if (intake_outcome === 'statute_expired') {
await sf.update('Lead', contact.sfdc_id, { Status: 'Closed - Statute Expired' });
}
return { ok: true, ai_call_id: aiCall.Id, outcome: intake_outcome };
